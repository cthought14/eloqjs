#include <ctype.h>
#include <stdio.h>
#include <string.h>

void reverse_name(char *name);

int main(void)
{
    char name[] = "Marijn Haverbeke";
    reverse_name(name);
    printf("%s\n", name);
}

void reverse_name(char *name)
{
    char *p = name;
    // Find first non-space.
    while (*p && isspace(*p))
        p++;
    if (!*p) return;
    // Found the initial.
    char initial = *p;
    // Find end of word.
    while (*p && !isspace(*p))
        p++;
    // Find beginning of word.
    while (*p && isspace(*p))
        p++;
    if (!*p) return;
    char *last_beg = p; // Beginning of last name.
    // Find end of word.
    while (*p && !isspace(*p))
        p++;
    char *last_end = p; // End of last name.
    // Move last name (last_end - last_beg) to beginning (name).
    memmove(name, last_beg, last_end - last_beg);
    p = name;
    // Move to end of last name.
    p += last_end - last_beg;
    // Add ", ".
    *p++ = ',';
    *p++ = ' ';
    // Add initial and ".".
    *p++ = initial;
    *p++ = '.';
    // Done.
    *p = '\0';
}
